{{- /* Reusable multi-source FullCalendar widget. Params (dict): page (required) the Page rendering this widget — used to flag that calendar assets are needed. id DOM id of the calendar container (default "calendar"). Must be unique per page. sources slice of source dicts, each: id (required, unique) FullCalendar event-source id; also used for toggle wiring. name (required) label shown in the legend / subscribe button. url (required) feed URL (FullCalendar JSON). Run through absURL. color (optional) hex color for this source's events. textColor (optional) hex text color for this source's events. subscribe (optional) ICS URL; renders a subscribe action for this source. default (optional, default true) whether the source starts visible / checked. initialView FullCalendar initial view (default "listYear"). showToggles render a legend of checkboxes that toggle each source on/off (default false). Asset loading is handled by calendarAssets.head.html (gated in head.html) and calendarAssets.scripts.html (gated in scripts.html on the Store flag set just below). */ -}} {{- $page := .page -}} {{- $id := .id | default "calendar" -}} {{- $initialView := .initialView | default "listYear" -}} {{- $showToggles := .showToggles | default false -}} {{- /* Signal the asset partials (read at end of body via Page.Store) that this page hosts a calendar. */ -}} {{- $page.Store.Set "needsCalendarAssets" true -}} {{- /* Normalize sources once; legend + JS both consume this. */ -}} {{- $jsSources := slice -}} {{- range .sources -}} {{- $sub := .subscribe | default "" -}} {{- if ne $sub "" }}{{ $sub = absURL $sub }}{{ end -}} {{- $def := true -}} {{- if isset . "default" }}{{ $def = .default }}{{ end -}} {{- $jsSources = $jsSources | append (dict "id" .id "name" .name "url" (absURL .url) "color" (.color | default "") "textColor" (.textColor | default "") "subscribe" $sub "default" $def) -}} {{- end -}} {{- if $showToggles }}
{{- range $jsSources }} {{- if .subscribe }} (subscribe){{ end }} {{- end }}
{{- end }}
